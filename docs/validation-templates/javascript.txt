const Ajv = require("ajv");
const fs = require("fs");

const schema = JSON.parse(fs.readFileSync("{{SCHEMA_FILENAME}}"));
const data = JSON.parse(fs.readFileSync("{{DATA_FILENAME}}"));

const ajv = new Ajv();
const validate = ajv.compile(schema);
const valid = validate(data);

if (!valid) {
  console.error(validate.errors);
} else {
  console.log("Valid!");
}
